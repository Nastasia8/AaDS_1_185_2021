from collections import deque
a,b= map(int,input().split())
nums=list(map(int,input().split()))
deq=deque()
for i in range(b):
  while deq and nums[i] <nums[deq[-1]]:
    deq.pop()
  deq.append(i)
print(nums[deq[0]])
for i in range(b,a):
  while deq and deq[0] <=i-b:
    deq.popleft()
  while deq and nums[i] <= nums[deq[-1]]:
    deq.pop()
  deq.append(i)
  print(nums[deq[0]])
